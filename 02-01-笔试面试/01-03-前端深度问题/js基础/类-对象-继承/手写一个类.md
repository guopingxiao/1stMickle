模块模式 用闭包模拟权限访问
function Egg() {
    var index = 0
    var data = [1,2,3,4,5]
    var length = data.length

    return {
        next: function () {
            var element;
            if (!this.hasNext()) {
                return null
            }
            element = data[index]
            index = index + 2
            return element

        },
        hasNext: function () {
            return index < length
        },
        rewind: function () {
            //            重置指针到初始位置
            index = 0
        },
        current: function () {
            // 不前进指针
            return data[index]
        }
    }
}
